Project launching:
    AutoStart:
        /etc/systemd/system/project.service:

            [Unit]
            Description=project
            After=network.target

            [Service]
            ExecStart=/env/project.sh
            Restart=always
            RestartSec=10

            [Install]
            WantedBy=multi-user.target

        sudo systemctl daemon-reload
        sudo systemctl enable project.service
    project.sh:

        #!bin/bash
        sleep 10
        source /env/venv/bin/activate
        export PYTHONPATH="$PYTHONPATH:/env/venv/lib/python3.13/site-packages"
        export PYTHONPATH="$PYTHONPATH:/env/core/core"
        python /env/projects/project/main.py

    ERRORS:
        203/EXEC - Права
    To DELETE AutoStart:
        sudo systemctl stop project.service
        sudo systemctl disable project.service
        sudo rm /etc/systemd/system/project.service
        sudo systemctl daemon-reload


Settings Apache2:
    /etc/apache2/sites-available/project.conf:

        <VirtualHost *:80>
            ServerAdmin webmaster@localhost
            DocumentRoot /env/projects/telegram_text_bot/static
            ServerName server_IP
            
            ProxyPass / http://localhost:8000/
            ProxyPassReverse / http://localhost:8000/
        </VirtualHost>

    sudo a2ensite project.conf
    sudo systemctl restart apache2
